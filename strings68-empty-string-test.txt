print "=== Test 1: Check if string is empty ==="

1 = ""
if 1 == "" then
    print "Variable 1 is empty"
endif

2 = "not empty"
if 2 == "" then
    print "Variable 2 is empty"
else
    print "Variable 2 is not empty"
endif

print "=== Test 2: Check if string is NOT empty ==="

3 = ""
if 3 != "" then
    print "Variable 3 is not empty"
else
    print "Variable 3 is empty"
endif

4 = "has content"

    print "Variable 4 is not empty"
endif

print "=== Test 7: Building from empty ==="

50 = ""
51 = "0"
while 50 != "xxx" do
    if 51 == "0" then
        print "x"
        50 = "x"
        51 = "1"
    else
        if 51 == "1" then
            print "xx"
            50 = "xx"
            51 = "2"
        else
            if 51 == "2" then
                print "xxx"
                50 = "xxx"
            endif
        endif
    endif
endwhile
print "Final:"
print 50
